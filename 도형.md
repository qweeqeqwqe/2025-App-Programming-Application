void main(){
  var n = 5;
  var result = "";
  for(var y = 0; y < n; y++){
    for ( var x = 0; x < n; x++){
      var c = 
        y == 0 || y == n - 1 ||
        x == 0 || x == n - 1;
      if (c) {
        result += "=";
      }
      else {
        result += " ";
      }
    }
    result += "\n";
  }
  print(result);
  
  
  
}
